ES6: const, let

Global variables and var are not allowed anymore

a = 'foo'; // DO NOT USE THIS
var a = 'foo'; // DO NOT USE THIS

Always use const and let

const a = 'foo';
a = 'bar'; // NOT ALLOWED -> COMPILER ERROR
let b = 'horse';
b = 'radish';

const options = {foo: 'bar', 'pi': 3.14 }
const foo = `Config options are:

pi: ${Math.floor(options.pi)}
`;

        this._mydata = {foo: 2};

    constructor() {

ES6: spread operator

function doSomething(a, b, c) {
    // do something with a, b, and c
}
const params = ['foo', 1.5, 'bar'];
doSomethig(...params)

const newList = [1, 2, 1]
const combinedList = [...newList, ...params]
// combinedList is [1, 2, 1, 'foo', 1.5, 'bar']

}

TestableClass.isTestable // true

ES7: decorators

function testableDecorator(target) {
    return (options) => {
        target.isTestable = true;
        target.options = options;
    };
}

@testableDecorator({
    styles: '.main { color: red; }'
})
class TestableClass {
}

TestableClass.isTestable // true
TestableClass.options.styles // '.main { color: red; }'
